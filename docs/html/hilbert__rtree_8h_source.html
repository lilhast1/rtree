321 void adjust_mbr() {
322 if (entries.empty()) {
323 mbr = Rectangle(Point(dims, 0), Point(dims, 0));
324 return;
325 }
326
327 Point lo(dims, INT64_MAX);
328 Point hi(dims, INT64_MIN);
329
330 for (auto& entry : entries) {
331 auto& rect = entry->get_mbr();
332 for (int i = 0; i < dims; i++) {
333 if (rect.lower[i] < lo[i])
334 lo[i] = rect.lower[i];
335 if (rect.higher[i] > hi[i])
336 hi[i] = rect.higher[i];

350 std::deque<Node<T>*> get_siblings(size_t num) {
351 std::deque<Node<T>*> result;
352 std::set<Node<T>*> visited;
353
354 result.push_back(this);
355 visited.insert(this);
356
357 auto right = next_sibling;
358 size_t iterations = 0;
359 const size_t MAX_ITERATIONS = 10000;
360
361 while (result.size() < num && right != nullptr && iterations < MAX_ITERATIONS) {
362 iterations++;
363
364 // Check for obvious corruption
365 if (right == this) {
366 break; // Self-loop
367 }
368
369 if (visited.find(right) != visited.end()) {
370 break; // Cycle detected
371 }
372
373 // Verify parent relationship - siblings should have same parent
374 if (right->get_parent() != this->get_parent()) {
375 break; // Different parent - stop here
376 }
377
378 result.push_back(right);
379 visited.insert(right);
380 right = right->next_sibling;
381 }
382
383 return result;
384 }

445 void insert(const Rectangle& rect, T* elem) {
446 ll h = curve.index(rect.get_center());
447 if (this->root == nullptr) {
448 this->root = new Node<T>(min_entries, max_entries, curve);
449 all_nodes.insert(root);
450 this->root->set_leaf(true);

453 std::deque<Node<T>*> out_siblings;

459 L->insert_leaf_entry(newEntry);
460 L->adjust_lhv();
461 L->adjust_mbr();
462 out_siblings.push_back(L);
463 } else {
464 NN = handle_overflow(L, newEntry, out_siblings);
465 }
466

509 void fix_cross_parent_sibling_links(std::deque<Node<T>*>& siblings) {
510 if (siblings.empty() || siblings[0]->is_leaf())
511 return;
512
513 for (size_t i = 0; i < siblings.size() - 1; i++) {
514 Node<T>* current_parent = siblings[i];
515 Node<T>* next_parent = siblings[i + 1];
516
517 if (current_parent->get_entries().empty() || next_parent->get_entries().empty())
518 continue;
519
520 auto last_entry_it = current_parent->get_entries().end();
521 --last_entry_it;
522 auto last_inner = dynamic_cast<InnerNode<T>*>(*last_entry_it);
523 if (!last_inner)
524 continue;
525 Node<T>* last_child = last_inner->get_node();
526
527 auto first_entry_it = next_parent->get_entries().begin();
528 auto first_inner = dynamic_cast<InnerNode<T>*>(*first_entry_it);
529 if (!first_inner)
530 continue;
531 Node<T>* first_child = first_inner->get_node();
532
533 // CRITICAL: Prevent cycles
534 if (last_child == first_child || last_child == current_parent
535 || first_child == next_parent || last_child->get_parent() == first_child
536 || first_child->get_parent() == last_child) {
537 continue;
538 }
539
540 last_child->set_next_siblings(first_child);
541 first_child->set_prev_siblings(last_child);
542 }

545 void redistribute_entries(EntrySet<T>& entries, std::deque<Node<T>*>& siblings) {
546 if (siblings.empty() || entries.empty()) {
547 return;
548 }
549
550 size_t batch_size = (entries.size() + siblings.size() - 1) / siblings.size();
551 auto siblings_it = siblings.begin();
552 size_t current_batch = 0;
553
554 // Create vector for iteration, clear set to prevent reuse
555 std::vector<NodeEntry<T>*> entries_vec(entries.begin(), entries.end());
556 entries.clear();
557
558 for (auto entry : entries_vec) {
559 if (entry->is_leaf()) {
560 (*siblings_it)->insert_leaf_entry(dynamic_cast<LeafEntry<T>*>(entry));
561 } else {
562 auto inner = dynamic_cast<InnerNode<T>*>(entry);
563 // Direct insert without going through insert_inner_entry to avoid sibling link
564 // updates
565 (*siblings_it)->entries.insert(inner);
566 inner->get_node()->set_parent(*siblings_it);
567 }
568
569 current_batch++;
570
571 if (current_batch >= batch_size && siblings_it != siblings.end()) {
572 (*siblings_it)->adjust_lhv();
573 (*siblings_it)->adjust_mbr();
574 ++siblings_it;
575 current_batch = 0;
576
577 if (siblings_it == siblings.end()) {
578 --siblings_it;
579 }
580 }
581 }
582
583 // Adjust last batch
584 if (siblings_it != siblings.end()) {
585 (*siblings_it)->adjust_lhv();
586 (*siblings_it)->adjust_mbr();
587 }
588
589 // Fix child sibling links within each parent
590 for (auto node : siblings) {
591 if (!node->is_leaf() && !node->get_entries().empty()) {
592 fix_child_sibling_links(node);
593 }
594 }
595
596 // Fix cross-parent links
597 fix_cross_parent_sibling_links(siblings);
598 }

635 Node<T>* handle_overflow(Node<T>* target, NodeEntry<T>* entry,
636 std::deque<Node<T>*>& out_siblings) {
637 EntrySet<T> entries;
638 Node<T>* newNode = nullptr;
639
640 out_siblings = target->get_siblings(2);
641
642 Node<T>* leftmost_external_prev = out_siblings.front()->get_prev_siblings();
643 Node<T>* rightmost_external_next = out_siblings.back()->get_next_siblings();
644 Node<T>* original_parent = target->get_parent();
645
646 entries.insert(entry);
647
648 for (auto node : out_siblings) {
649 entries.insert(node->get_entries().begin(), node->get_entries().end());
650 node->reset_entries();
651 node->set_prev_siblings(nullptr);
652 node->set_next_siblings(nullptr);
653 }
654
655 if (entries.size() > out_siblings.size() * static_cast<size_t>(max_entries)) {
656 newNode = new Node<T>(min_entries, max_entries, curve);
657 all_nodes.insert(newNode);
658 newNode->set_leaf(entry->is_leaf());
659 newNode->set_parent(original_parent);
660 out_siblings.push_back(newNode);
661 }
662
663 redistribute_entries(entries, out_siblings);
664
665 for (auto node : out_siblings) {
666 node->set_parent(original_parent);
667 }
668
669 // Set up sibling chain for redistributed nodes
670 for (size_t i = 0; i < out_siblings.size(); ++i) {
671 if (i > 0) {
672 out_siblings[i]->set_prev_siblings(out_siblings[i - 1]);
673 } else {
674 out_siblings[i]->set_prev_siblings(leftmost_external_prev);
675 }
676
677 if (i < out_siblings.size() - 1) {
678 out_siblings[i]->set_next_siblings(out_siblings[i + 1]);
679 } else {
680 out_siblings[i]->set_next_siblings(rightmost_external_next);
681 }
682 }
683
684 // Update external neighbors to point to the new chain
685 if (leftmost_external_prev != nullptr) {
686 // Make sure we're not creating a cycle
687 if (leftmost_external_prev != out_siblings.front()) {
688 leftmost_external_prev->set_next_siblings(out_siblings.front());
689 }
690 }
691
692 if (rightmost_external_next != nullptr) {
693 // Make sure we're not creating a cycle
694 if (rightmost_external_next != out_siblings.back()) {
695 rightmost_external_next->set_prev_siblings(out_siblings.back());
696 }
697 }
698
699 return newNode;
700 }

702 Node<T>* handle_underflow(Node<T>* target, std::deque<Node<T>*>& out_siblings) {
703 if (target == nullptr) {
704 return nullptr;
705 }
706
707 out_siblings = target->get_siblings(3);
708
709 if (out_siblings.empty()) {
710 out_siblings.push_back(target);
711 return nullptr;
712 }
713
714 // If we got fewer siblings than requested due to a cycle,
715 // we can't safely redistribute - just return target as-is
716 if (out_siblings.size() < 2) {
717 return nullptr;
718 }
719
720 // STEP 1: Save external neighbors BEFORE touching anything
721 Node<T>* leftmost_external_prev = out_siblings.front()->get_prev_siblings();
722 Node<T>* rightmost_external_next = out_siblings.back()->get_next_siblings();
723
724 // Verify these are actually external (not in our sibling group)
725 std::set<Node<T>*> sibling_set(out_siblings.begin(), out_siblings.end());
726 if (leftmost_external_prev != nullptr
727 && sibling_set.find(leftmost_external_prev) != sibling_set.end()) {
728 leftmost_external_prev = nullptr; // Not external, it's in our group
729 }
730 if (rightmost_external_next != nullptr
731 && sibling_set.find(rightmost_external_next) != sibling_set.end()) {
732 rightmost_external_next = nullptr; // Not external, it's in our group
733 }
734
735 Node<T>* original_parent = target->get_parent();
736
737 // STEP 2: Collect ALL entries before making any changes
738 EntrySet<T> entries;
739 for (auto node : out_siblings) {
740 entries.insert(node->get_entries().begin(), node->get_entries().end());
741 }
742
743 // CRITICAL: If no entries collected, something is wrong - abort
744 if (entries.empty()) {
745 std::cerr << "ERROR: No entries collected from siblings in underflow\n";
746 return nullptr;
747 }
748
749 // STEP 3: Decide if we should remove a node
750 Node<T>* removed = nullptr;
751 bool should_remove =
752 (out_siblings.size() > 1) && original_parent != nullptr
753 && entries.size() < out_siblings.size() * static_cast<size_t>(min_entries);
754
755 if (should_remove) {
756 removed = out_siblings.front();
757 out_siblings.pop_front();
758
759 // Update left external if the removed node was connected to it
760 if (removed->get_prev_siblings() == leftmost_external_prev) {
761 leftmost_external_prev = removed->get_prev_siblings();
762 }
763 }
764
765 // STEP 3b: After removal, check if we still have enough nodes
766 if (out_siblings.empty()) {
767 std::cerr << "ERROR: All siblings removed in underflow\n";
768 if (removed != nullptr) {
769 out_siblings.push_back(removed);
770 removed = nullptr;
771 }
772 return nullptr;
773 }
774
775 // STEP 4: Completely disconnect all nodes in the group
776 for (auto node : out_siblings) {
777 node->reset_entries();
778 node->set_prev_siblings(nullptr);
779 node->set_next_siblings(nullptr);
780 }
781
782 if (removed != nullptr) {
783 removed->reset_entries();
784 removed->set_prev_siblings(nullptr);
785 removed->set_next_siblings(nullptr);
786 }
787
788 // Also disconnect external neighbors from us
789 if (leftmost_external_prev != nullptr) {
790 leftmost_external_prev->set_next_siblings(nullptr);
791 }
792 if (rightmost_external_next != nullptr) {
793 rightmost_external_next->set_prev_siblings(nullptr);
794 }
795
796 // STEP 5: Redistribute entries
797 redistribute_entries(entries, out_siblings);
798
799 // STEP 6: Set parent
800 for (auto node : out_siblings) {
801 if (node != nullptr) {
802 node->set_parent(original_parent);
803 }
804 }
805
806 // STEP 7: Build clean internal chain
807 for (size_t i = 0; i < out_siblings.size(); ++i) {
808 if (i > 0) {
809 out_siblings[i]->set_prev_siblings(out_siblings[i - 1]);
810 } else {
811 out_siblings[i]->set_prev_siblings(nullptr);
812 }
813
814 if (i < out_siblings.size() - 1) {
815 out_siblings[i]->set_next_siblings(out_siblings[i + 1]);
816 } else {
817 out_siblings[i]->set_next_siblings(nullptr);
818 }
819 }
820
821 // STEP 8: Reconnect to external neighbors
822 if (!out_siblings.empty()) {
823 if (leftmost_external_prev != nullptr) {
824 out_siblings.front()->set_prev_siblings(leftmost_external_prev);
825 leftmost_external_prev->set_next_siblings(out_siblings.front());
826 }
827
828 if (rightmost_external_next != nullptr) {
829 out_siblings.back()->set_next_siblings(rightmost_external_next);
830 rightmost_external_next->set_prev_siblings(out_siblings.back());
831 }
832 }
833
834 return removed;
835 }

837 Node<T>* adjust_tree(Node<T>* root, Node<T>* N, Node<T>* NN, std::deque<Node<T>*>& siblings) {
838 Node<T>* PP = nullptr;
839 auto newRoot = root;
840 std::deque<Node<T>*> newSiblings;
841 std::set<Node<T>*> P;
842 std::set<Node<T>*> S;
843 S.insert(siblings.begin(), siblings.end());
844
845 int iteration = 0;
846
847 while (true) {
848 iteration++;
849 if (iteration > 1000) {
850 std::cerr << "ERROR: adjust_tree infinite loop detected!\n";
851 throw std::runtime_error("Infinite loop in adjust_tree");
852 }
853
854 Node<T>* Np = N->get_parent();
855
856 if (Np == nullptr) {
857 if (NN != nullptr) {
858 auto n = new InnerNode<T>(N);
859 auto nn = new InnerNode<T>(NN);
860
861 newRoot = new Node<T>(min_entries, max_entries, curve);
862 all_nodes.insert(newRoot);
863 newRoot->insert_inner_entry(n);
864 newRoot->insert_inner_entry(nn);
865
866 // CRITICAL: Fix the new root's child chain
867 validate_and_fix_child_chain(newRoot);
868 }
869
870 newRoot->adjust_lhv();
871 newRoot->adjust_mbr();
872 break;
873 } else {
874 if (NN != nullptr) {
875 auto nn = new InnerNode<T>(NN);
876
877 if (Np->entries.size() < static_cast<size_t>(max_entries)) {
878 Np->insert_inner_entry(nn);
879 // CRITICAL: Fix parent's child chain after insertion
880 validate_and_fix_child_chain(Np);
881 Np->adjust_lhv();
882 Np->adjust_mbr();
883 newSiblings.push_back(Np);
884 } else {
885 PP = handle_overflow(Np, nn, newSiblings);
886 // After overflow, validate all affected parents
887 for (auto sibling : newSiblings) {
888 if (sibling && !sibling->is_leaf()) {
889 validate_and_fix_child_chain(sibling);
890 }
891 }
892 }
893 } else {
894 newSiblings.push_back(Np);
895 }
896
897 for (auto node : S) {
898 if (node != nullptr && node->get_parent() != nullptr) {
899 P.insert(node->get_parent());
900 }
901 }
902
903 for (auto node : P) {
904 if (node != nullptr) {
905 node->adjust_lhv();
906 node->adjust_mbr();
907 // CRITICAL: Validate child chains after adjustments
908 if (!node->is_leaf()) {
909 validate_and_fix_child_chain(node);
910 }
911 }
912 }
913
914 N = Np;
915 NN = PP;
916 PP = nullptr;
917
918 S.clear();
919 P.clear();
920
921 S.insert(newSiblings.begin(), newSiblings.end());
922 newSiblings.clear();
923 }
924 }
925
926 return newRoot;

929 void condense_tree(Node<T>* node, Node<T>* del_node, std::deque<Node<T>*>& siblings) {
930 std::deque<Node<T>*> new_siblings;
931 std::set<Node<T>*> s, p;
932 s.insert(siblings.begin(), siblings.end());
933
934 int safety_counter = 0;
935 const int MAX_ITERATIONS = INT_MAX;
936
937 while (true) {
938 safety_counter++;
939 if (safety_counter > MAX_ITERATIONS) {
940 std::cerr << "ERROR: Too many iterations in condense_tree\n";
941 break;
942 }
943
944 auto np = node->get_parent();
945 Node<T>* dpparent = nullptr;
946
947 if (np == nullptr) {
948 // Root condensation
949 if (!node->is_leaf() && node->get_entries().size() == 1) {
950 auto main_entry = dynamic_cast<InnerNode<T>*>(*node->get_entries().begin());
951 if (main_entry) {
952 auto main_node = main_entry->get_node();
953
954 // Verify main_node is still valid
955 if (main_node != nullptr && all_nodes.find(main_node) != all_nodes.end()) {
956 EntrySet<T> data;
957 data.insert(main_node->get_entries().begin(),
958 main_node->get_entries().end());
959
960 main_node->reset_entries();
961 node->reset_entries();
962
963 if (main_node->is_leaf()) {
964 node->set_leaf(true);
965 for (auto entry : data) {
966 node->insert_leaf_entry(dynamic_cast<LeafEntry<T>*>(entry));
967 }
968 } else {
969 for (auto entry : data) {
970 node->insert_inner_entry(dynamic_cast<InnerNode<T>*>(entry));
971 }
972 // Fix child sibling chain after moving entries to root
973 validate_and_fix_child_chain(node);
974 }
975
976 all_nodes.erase(main_node);
977 delete main_entry;
978 delete main_node;
979 }
980 }
981 }
982
983 if (!node->get_entries().empty()) {
984 node->adjust_lhv();
985 node->adjust_mbr();
986 }
987 break;
988 } else {
989 // Handle deleted node
990 if (del_node != nullptr) {
991 auto dnparent = del_node->get_parent();
992
993 // CRITICAL: Check if node still exists before accessing
994 bool node_exists = (all_nodes.find(del_node) != all_nodes.end());
995
996 if (node_exists && dnparent != nullptr
997 && all_nodes.find(dnparent) != all_nodes.end()) {
998 // Remove from parent
999 dnparent->remove_inner_entry(del_node);
1000
1001 // CRITICAL: Fix parent's child sibling chain after removal
1002 validate_and_fix_child_chain(dnparent);
1003
1004 if (dnparent->underflow() && dnparent->get_parent() != nullptr) {
1005 dpparent = handle_underflow(dnparent, new_siblings);
1006 } else {
1007 new_siblings.push_back(dnparent);
1008 }
1009 }
1010
1011 // CRITICAL: Clear all pointers TO the deleted node before freeing it
1012 auto del_prev = del_node->get_prev_siblings();
1013 auto del_next = del_node->get_next_siblings();
1014
1015 if (del_prev != nullptr && all_nodes.find(del_prev) != all_nodes.end()) {
1016 del_prev->set_next_siblings(del_next);
1017 }
1018 if (del_next != nullptr && all_nodes.find(del_next) != all_nodes.end()) {
1019 del_next->set_prev_siblings(del_prev);
1020 }
1021
1022 del_node->set_prev_siblings(nullptr);
1023 del_node->set_next_siblings(nullptr);
1024
1025 // Delete the node if it still exists
1026 // (handle_underflow may have already erased it from all_nodes)
1027 if (node_exists && all_nodes.find(del_node) != all_nodes.end()) {
1028 all_nodes.erase(del_node);
1029 }
1030
1031 // Always safe to delete - either it's still valid or already erased
1032 delete del_node;
1033 del_node = nullptr;
1034 }
1035
1036 if (np != nullptr && all_nodes.find(np) != all_nodes.end()) {
1037 new_siblings.push_back(np);
1038 }
1039
1040 // Collect parents of affected siblings
1041 for (auto snode : s) {
1042 if (snode != nullptr && all_nodes.find(snode) != all_nodes.end()
1043 && snode->get_parent() != nullptr
1044 && all_nodes.find(snode->get_parent()) != all_nodes.end()) {
1045 p.insert(snode->get_parent());
1046 }
1047 }
1048
1049 // Adjust MBR and LHV for all affected parents
1050 for (auto pnode : p) {
1051 if (pnode != nullptr && all_nodes.find(pnode) != all_nodes.end()
1052 && !pnode->get_entries().empty()) {
1053 pnode->adjust_lhv();
1054 pnode->adjust_mbr();
1055 // CRITICAL: Validate child chains after adjustments
1056 if (!pnode->is_leaf()) {
1057 validate_and_fix_child_chain(pnode);
1058 }
1059 }
1060 }
1061
1062 node = np;
1063 del_node = dpparent;
1064 s.clear();
1065 p.clear();
1066 s.insert(new_siblings.begin(), new_siblings.end());
1067 new_siblings.clear();
1068 }
1069 }
