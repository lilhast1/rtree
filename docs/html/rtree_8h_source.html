51 [[nodiscard]] double area() const {
52 double a = 1;

78 template <typename Itr>
79 static Rectangle calc_mbr(Itr p, Itr q) {
80 if (q == p)
81 throw std::range_error("Calculating the MBR on an empty range");
82 auto t = p++;
83 if (q == p)
84 return (*t)->mbr;
85 auto n = p;
86 auto agg = calc_mbr((*n)->mbr, (*t)->mbr);
87 while (n != q) {
88 agg = calc_mbr(agg, (*n)->mbr);
89 n++;
90 }
91 return agg;
92 }

94 static Rectangle calc_mbr(const Rectangle& a, const Rectangle& b) {
95 auto c = a;
96 std::transform(a.min.begin(), a.min.end(), b.min.begin(), c.min.begin(),
97 [](double a, double b) { return std::min(a, b); });
98 std::transform(a.max.begin(), a.max.end(), b.max.begin(), c.max.begin(),
99 [](double a, double b) { return std::max(a, b); });
100 return c;
101 }

104 return calc_mbr(a, b).area();

281 void condense_tree(Node<T>* l) {
282 Node<T>* n = l;
283 std::vector<Node<T>*> q; // Set of eliminated nodes to reinsert
284
285 while (n != root) {
286 Node<T>* p = n->parent;
287
288 if (n->count() < m) {
289 // Remove n from parent's children list
290 auto it = std::find(p->children.begin(), p->children.end(), n);
291 if (it != p->children.end()) {
292 p->children.erase(it);
293 }
294
295 // CRITICAL: Set parent to nullptr to mark this node as orphaned
296 n->parent = nullptr;
297
298 // Add to elimination list
299 q.push_back(n);
300 } else {
301 // Node is not underflow, just update its MBR
302 update_mbr(n);
303 }
304
305 // Always move to parent
306 n = p;
307 }
308
309 // Update root MBR if it still exists and has children
310 if (root && root->count() > 0) {
311 update_mbr(root);
312 }
313
314 // Reinsert all entries from eliminated nodes
315 std::vector<std::pair<T*, Rectangle>> leaf_entries;
316
317 for (Node<T>* orphan : q) {
318 if (orphan->is_leaf) {
319 // Collect leaf entries
320 for (auto& elem : orphan->elems) {
321 leaf_entries.push_back(elem);
322 }
323 // Clear to prevent destructor issues
324 for (auto& elem : orphan->elems) {
325 elem.first = nullptr;
326 }
327 orphan->elems.clear();
328 delete orphan;
329 } else {
330 // For non-leaf nodes, collect all leaf entries from subtree
331 collect_data_from_subtree(orphan, leaf_entries);
332 // Destructor will handle cleanup
333 delete orphan;
334 }
335 }
336
337 // Reinsert all collected entries
338 for (const auto& entry : leaf_entries) {
339 insert(entry.second, entry.first);

433 Node<T>* split(Node<T>* t) {
434 // apply pickseeds
435 std::vector<EntryWrapper> entries;
436 std::vector<bool> assigned(t->count(), false);
437 if (t->is_leaf) {
438 for (int i = 0; i < t->count(); i++)
439 entries.push_back(EntryWrapper{i, t->elems[i].second});
440 } else {
441 for (int i = 0; i < t->count(); i++)
442 entries.push_back(EntryWrapper{i, t->children[i]->mbr});
443 }
444
445 auto seeds = pick_seeds(entries);
446 assigned[seeds.first] = assigned[seeds.second] = true;
447
448 int n = t->count();
449
450 auto mbr1 = entries[seeds.first].rect;
451 auto mbr2 = entries[seeds.second].rect;
452
453 std::vector<int> g1{seeds.first}, g2{seeds.second};
454
455 for (int i = 0; i < n; i++) {
456 int next = pick_next(entries, assigned, mbr1, mbr2);
457 if (next < 0)
458 continue;
459 assigned[next] = true;
460 auto c1 = Rectangle::calc_mbr(mbr1, entries[next].rect);
461 auto c2 = Rectangle::calc_mbr(mbr2, entries[next].rect);
462 if (c1.area() < c2.area()) {
463 mbr1 = c1;
464 g1.push_back(next);
465 } else {
466 mbr2 = c2;
467 g2.push_back(next);
468 }
469 }
470
471 // Enforce minimum entries per node
472 if (g1.size() < m || g2.size() < m) {
473 auto& big = g1.size() > g2.size() ? g1 : g2;
474 auto& small = g1.size() < g2.size() ? g1 : g2;
475
476 while (small.size() < m) {
477 small.push_back(big.back());
478 big.pop_back();
479 }
480
481 // CRITICAL FIX: Recalculate MBRs after redistribution
482 mbr1 = entries[g1[0]].rect;
483 for (size_t i = 1; i < g1.size(); i++) {
484 mbr1 = Rectangle::calc_mbr(mbr1, entries[g1[i]].rect);
485 }
486
487 mbr2 = entries[g2[0]].rect;
488 for (size_t i = 1; i < g2.size(); i++) {
489 mbr2 = Rectangle::calc_mbr(mbr2, entries[g2[i]].rect);
490 }
491 }
492
493 auto tt = new Node<T>{t->is_leaf, mbr2};
494 tt->parent = t->parent;
495 t->mbr = mbr1;
496
497 if (t->is_leaf) {
498 std::vector<std::pair<T*, Rectangle>> elems1;
499 std::vector<std::pair<T*, Rectangle>> elems2;
500
501 for (auto i : g1) elems1.push_back(t->elems[i]);
502 for (auto i : g2) elems2.push_back(t->elems[i]);
503
504 for (auto& e : t->elems) {
505 e.first = nullptr;
506 }
507
508 t->elems = elems1;
509 tt->elems = elems2;
510 } else {
511 std::vector<Node<T>*> elems1;
512 std::vector<Node<T>*> elems2;
513
514 for (auto i : g1) elems1.push_back(t->children[i]);
515 for (auto i : g2) elems2.push_back(t->children[i]);
516
517 for (auto& e : t->children) {
518 e = nullptr;
519 }
520
521 t->children = elems1;
522 tt->children = elems2;
523
524 // Update parent pointers
525 for (auto child : t->children) {
526 child->parent = t;
527 }
528 for (auto child : tt->children) {
529 child->parent = tt;
530 }
531 }
532
533 return tt;
534 }

560 std::pair<int, int> pick_seeds(const std::vector<EntryWrapper>& entries) {
561 double max_waste = -1;
562 int n = entries.size();
563 std::pair<int, int> seeds{-1, -1};
564 for (int i = 0; i < n; i++) {
565 for (int j = 0; j < n; j++) {
566 if (i == j)
567 continue;
568 auto b = Rectangle::calc_mbr(entries[i].rect, entries[j].rect);
569 double d = b.area() - entries[i].rect.area() - entries[j].rect.area();
570 if (std::fabs(d) > max_waste) {
571 max_waste = d;
572 seeds.first = i;
573 seeds.second = j;
574 }
575 }
576 }
577 return seeds;

639 Node<T>* n = l;
640 std::vector<Node<T>*> orphans;
641
642 // 1. ASCEND AND COLLECT
643 while (n != root) {
644 Node<T>* p = n->parent;
645 if (n->count() < m) {
646 auto it = std::find(p->children.begin(), p->children.end(), n);
647 if (it != p->children.end())
648 p->children.erase(it);
649 n->parent = nullptr;
650 orphans.push_back(n);
651 } else {
652 n->update_mbr();
653 }
654 n = p;
655 }
656
657 // 2. ROOT SANITY CHECK (Fixes Extreme Condense/Massive Delete)
658 // If root was internal but all children were removed, it is now invalid.
659 // Reset it to an empty leaf.
660 if (!root->is_leaf && root->children.empty()) {
661 root->is_leaf = true;
662 root->elems.clear(); // Ensure clean state
663 }
664
665 root->update_mbr();
666
667 // 3. RE-INSERT ORPHANS
668 for (Node<T>* orphan : orphans) {
669 if (orphan->is_leaf) {
670 // Re-insert Data
671 for (auto& entry : orphan->elems) {
672 insert(entry.second, entry.first);
673 }
674 } else {
675 // Re-insert Subtrees (Internal Nodes)
676 for (Node<T>* child_subtree : orphan->children) {
677 insert_internal(child_subtree);
678 }
679 }
680 // Cleanup Orphan Container
681 orphan->children.clear();
682 orphan->elems.clear();
683 delete orphan;
684 }
685
686 // 4. ROOT HEIGHT REDUCTION
687 if (!root->is_leaf && root->children.size() == 1) {
688 Node<T>* new_root = root->children[0];
689 root->children.clear();
690 delete root;
691 root = new_root;
692 root->parent = nullptr;
693 }

714 Node<T>* choose_subtree(Node<T>* node, const Rectangle& rect, int insertion_level) {
715 // Base Case: If we are at the level immediately above the insertion level,
716 // this node is the candidate parent.
717 // Example: Inserting a Leaf (level 0). We need a parent at Level 1 (Internal).
718 // Note: You need a way to determine node height/level.
719 // If you don't store 'level' in Node, you calculate it based on distance to leaves.
720
721 int current_level = get_level(node); // See helper below
722 if (current_level == insertion_level + 1) {
723 return node;
724 }
725
726 // Standard R-Tree logic: Find child requiring least enlargement
727 Node<T>* best_child = nullptr;
728 double min_enlargement = std::numeric_limits<double>::max();
729 double best_area = std::numeric_limits<double>::max();
730
731 for (Node<T>* child : node->children) {
732 double area = child->mbr.area();
733 double enlargement = Rectangle::calc_mbr(child->mbr, rect).area() - area;
734
735 if (enlargement < min_enlargement) {
736 min_enlargement = enlargement;
737 best_area = area;
738 best_child = child;
739 } else if (enlargement == min_enlargement) {
740 if (area < best_area) {
741 best_area = area;
742 best_child = child;
743 }
744 }
745 }
746
747 // Recurse
748 return choose_subtree(best_child, rect, insertion_level);
749 }

772 Node<T>* choose_subtree_at_level(Node<T>* node, const Rectangle& mbr, int target_height) {
773 if (node->is_leaf)
774 return node; // Should not happen for internal insert, but safety first
775
776 // Guard against empty internal nodes (e.g. root temporarily empty)
777 if (node->children.empty())
778 return node;
779
780 // Optimization: Check height of immediate children
781 // If children are at the target height, then 'node' is the parent we want.
782 int child_height = get_height(node->children[0]);
783 if (child_height == target_height) {
784 return node;
785 }

787 // Standard Min-Enlargement Search
788 Node<T>* best_child = nullptr;
789 double min_enlargement = std::numeric_limits<double>::max();
790 double best_area = std::numeric_limits<double>::max();
791
792 for (Node<T>* child : node->children) {
793 double area = child->mbr.area();
794 double enlargement = Rectangle::calc_mbr(child->mbr, mbr).area() - area;
795
796 if (enlargement < min_enlargement) {
797 min_enlargement = enlargement;
798 best_area = area;
799 best_child = child;
800 } else if (enlargement == min_enlargement) {
801 if (area < best_area) {
802 best_area = area;
803 best_child = child;
804 }
805 }
806 }
807
808 // If for some reason we didn't find a child (shouldn't happen), return node
809 if (!best_child)

811
812 return choose_subtree_at_level(best_child, mbr, target_height);
813 }

816 void insert_internal(Node<T>* subtree) {
817 int subtree_height = get_height(subtree);
818 int root_height = get_height(root);
819
820 // CASE: Tree is too short.
821 // If the orphan is as tall as the root, we must create a new root
822 // to hold both the old root and the orphan.
823 if (subtree_height >= root_height) {
824 Node<T>* new_root = new Node<T>(false, root->mbr); // Internal
825 new_root->children.push_back(root);
826 root->parent = new_root;
827
828 // The subtree also goes into this new root (or its appropriate level,
829 // but if heights match, it goes here).
830 // However, to keep logic uniform, we just update root and let choose_subtree work.
831 root = new_root;
832 root->update_mbr();
833 }
834
835 // Find the correct parent (guaranteed to exist now)
836 Node<T>* target_parent = choose_subtree_at_level(root, subtree->mbr, subtree_height);
837
838 // Link
839 target_parent->children.push_back(subtree);
840 subtree->parent = target_parent;
841
842 // Update MBRs
843 target_parent->update_mbr();
844
845 // Check Overflow
846 if (target_parent->children.size() > M) {
847 // Assuming you have a standard split function
848 auto ll = split(target_parent);
849 adjust_tree(target_parent, ll);
850 } else {
851 // Propagate MBR changes up
852 Node<T>* p = target_parent->parent;
853 while (p) {
854 p->update_mbr();
855 p = p->parent;
856 }
857 }
858 }
